<script>
  (function () {
    var root = document.querySelector(".gmB04-root");
    if (!root) return;
    var ringHost = root.querySelector(".gmB04-peek-media");
    var popupHost = root.querySelector(".gmB04-peek-popup-media");
    var modal = root.querySelector("#gmB04PeekModal");
    if (!ringHost || !popupHost || !modal) return;
    var desktopPeekY = getComputedStyle(root).getPropertyValue("--gmB04-peek-y").trim() || "330px";

    function getVideoUrl() { return getComputedStyle(root).getPropertyValue("--peek-video-url").trim().replace(/^[\"']|[\"']$/g, ""); }
    function youtubeId(url) { var m = url.match(/(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/)([a-zA-Z0-9_-]{6,})/); return m ? m[1] : ""; }
    function makeMedia(url, ringMode) {
      var yid = youtubeId(url);
      if (yid) {
        var iframe = document.createElement("iframe");
        iframe.allow = "autoplay; encrypted-media; picture-in-picture";
        iframe.referrerPolicy = "strict-origin-when-cross-origin";
        iframe.allowFullscreen = true;
        iframe.src = ringMode
          ? "https://www.youtube.com/embed/" + yid + "?autoplay=1&mute=1&controls=0&loop=1&playlist=" + yid + "&rel=0&modestbranding=1&playsinline=1"
          : "https://www.youtube.com/embed/" + yid + "?autoplay=1&controls=1&rel=0&modestbranding=1&playsinline=1";
        return iframe;
      }
      var video = document.createElement("video");
      video.src = url;
      video.playsInline = true;
      if (ringMode) { video.autoplay = true; video.muted = true; video.loop = true; } else { video.autoplay = true; video.controls = true; }
      return video;
    }
    function mountRing(){ var url=getVideoUrl(); if(!url) return; ringHost.innerHTML=""; ringHost.appendChild(makeMedia(url,true)); }
    function mountPopup(){ var url=getVideoUrl(); if(!url) return; popupHost.innerHTML=""; popupHost.appendChild(makeMedia(url,false)); }
    function syncPopupToHash(){ if(location.hash==="#gmB04PeekModal") mountPopup(); else popupHost.innerHTML=""; }
    function placePeekOnMobile(){
      if(window.innerWidth>960){ root.style.setProperty("--gmB04-peek-y", desktopPeekY); return; }
      var media=root.querySelector(".gmB04-media"); if(!media) return;
      var blockRect=root.querySelector(".gmB04-block").getBoundingClientRect();
      var mediaRect=media.getBoundingClientRect();
      var centerY=(mediaRect.top+mediaRect.height/2)-blockRect.top;
      root.style.setProperty("--gmB04-peek-y", centerY + "px");
    }
    function lockHashScrollOnMobile(){ if(window.innerWidth>960) return; var x=window.scrollX,y=window.scrollY; requestAnimationFrame(function(){ window.scrollTo(x,y); }); }
    function openModalNoScroll(e){ if(e) e.preventDefault(); var x=window.scrollX,y=window.scrollY; modal.classList.add("is-open"); mountPopup(); requestAnimationFrame(function(){ window.scrollTo(x,y); }); }
    function closeModalNoScroll(e){ if(e) e.preventDefault(); var x=window.scrollX,y=window.scrollY; modal.classList.remove("is-open"); popupHost.innerHTML=""; requestAnimationFrame(function(){ window.scrollTo(x,y); }); }
    function bindModalHandlers(){ var opener=root.querySelector(".gmB04-peek"); var closeBtn=modal.querySelector(".gmB04-peek-close"); var backdrop=modal.querySelector(".gmB04-peek-backdrop"); if(opener) opener.addEventListener("click", openModalNoScroll); if(closeBtn) closeBtn.addEventListener("click", closeModalNoScroll); if(backdrop) backdrop.addEventListener("click", closeModalNoScroll); window.addEventListener("keydown", function(ev){ if(ev.key==="Escape") closeModalNoScroll(ev); }); }

    mountRing(); syncPopupToHash(); placePeekOnMobile(); bindModalHandlers();
    window.addEventListener("hashchange", syncPopupToHash);
    window.addEventListener("hashchange", lockHashScrollOnMobile);
    window.addEventListener("resize", placePeekOnMobile);
  })();
</script>
